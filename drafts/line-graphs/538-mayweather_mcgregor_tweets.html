Extract all_flames

hyped_bored <- hyped_bored %>% 
  mutate(
    flames = str_extract_all(string = emojis_used, pattern = "🔥")
  ) %>% 
  tidyr::unnest_wider(data = .,
    col = flames,
    names_sep = "_") %>%
  tidyr::unite(
    col = "all_flames",
    contains("flames_"),
    sep = " ",
    na.rm = TRUE,
    remove = TRUE
  )
select(hyped_bored, emojis_used, all_flames)
